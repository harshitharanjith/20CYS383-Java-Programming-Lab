CODE:

n=float(input())

if n>=0.9 and n<1:
    print("A")
elif n>=0.8 and n<0.9:
    print("B")
elif n>=0.7 and n<0.8:
    print("C")
elif n>=0.6 and n<0.7:
    print("D")
elif n<0.6:
    print("F")
else:
    print("Score is out of range")
